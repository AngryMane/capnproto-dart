Stream<Uint8List> deserializeStreamRaw(
  1. Stream<Uint8List> bytes, [
  2. MessageReaderOptions options = const MessageReaderOptions()
])

Like deserializeStream but yields the raw framed bytes for each message instead of a parsed MessageReader. Useful when the caller needs both the parsed content and the original bytes (e.g., to echo them back in an Unimplemented response).

options bounds how much a single declared message is allowed to make this method buffer before any of its content has actually arrived: MessageReaderOptions.maxSegments caps the declared segment count (read from the first 4 bytes alone) and MessageReaderOptions.traversalLimitInWords caps the declared total size, so a peer can't force unbounded buffering by framing a header that claims an enormous message.

Example

// Given the required message, schema, or raw-layout values:
final operation = MessageStream.deserializeStreamRaw;

static Stream<Uint8List> deserializeStreamRaw(
  Stream<Uint8List> bytes, [
  MessageReaderOptions options = const MessageReaderOptions(),
]) async* {
  final buffer = <int>[];
  // Bytes at the front of `buffer` already yielded as part of a complete
  // message, but not yet physically dropped. Compacted away once, right
  // before the next chunk is appended (below) — not with a `removeRange`
  // per drained message — so a chunk containing several complete messages
  // back-to-back (the common case for a busy stream) costs one O(buffered
  // bytes) shift total, not one per message.
  var consumed = 0;

  await for (final chunk in bytes) {
    if (consumed > 0) {
      buffer.removeRange(0, consumed);
      consumed = 0;
    }
    buffer.addAll(chunk);

    // Drain all complete messages from the buffer.
    while (true) {
      final available = buffer.length - consumed;
      // Need at least 4 bytes to read (numSegments − 1).
      if (available < 4) break;

      final numSegments = _readUint32LE(buffer, consumed) + 1;
      if (numSegments > options.maxSegments) {
        throw DecodeException(
          'message declares $numSegments segments, exceeding maxSegments '
          '(${options.maxSegments})',
        );
      }

      // Header size: (1 + numSegments) uint32s, padded to 8-byte boundary
      // when numSegments is even (so the count of uint32s is odd).
      final headerUint32s = 1 + numSegments + (numSegments.isEven ? 1 : 0);
      final headerBytes = headerUint32s * 4;

      if (available < headerBytes) break;

      // Sum segment sizes to find total message byte count.
      int totalDataBytes = 0;
      for (int i = 0; i < numSegments; i++) {
        totalDataBytes += _readUint32LE(buffer, consumed + (1 + i) * 4) * 8;
      }
      if (totalDataBytes ~/ 8 > options.traversalLimitInWords) {
        throw DecodeException(
          'message declares ${totalDataBytes ~/ 8} words, exceeding '
          'traversalLimitInWords (${options.traversalLimitInWords})',
        );
      }

      final totalBytes = headerBytes + totalDataBytes;
      if (available < totalBytes) break;

      final msgBytes = Uint8List.fromList(
        buffer.sublist(consumed, consumed + totalBytes),
      );
      consumed += totalBytes;

      yield msgBytes;
    }
  }

  if (buffer.length - consumed > 0) {
    throw DecodeException(
      'stream ended with ${buffer.length - consumed} bytes of an '
      'incomplete message',
    );
  }
}
